
#### Examples

**cmdline parse:**

```go
package main

import (
	"fmt"

	"github.com/whale-clouds/goutil/cliutil"
	"github.com/whale-clouds/goutil/dump"
)

func main() {
	args := cliutil.ParseLine(`./app top sub --msg "has multi words"`)
	dump.P(args)

	s := cliutil.BuildLine("./myapp", []string{
		"-a", "val0",
		"-m", "this is message",
		"arg0",
	})
	fmt.Println("Build line:", s)
}
```
